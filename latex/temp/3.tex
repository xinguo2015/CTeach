\begin{document}

  \begin{animateinline}[
    begin={
    \begin{tikzpicture}[blue,scale=1,line width=2pt]
    \useasboundingbox[draw] (0,0)rectangle(4,4);
    },
    end={
     \end{tikzpicture}
     }
     ]{2}

    \draw[->](0,0)--(1,1);\newframe
    \draw[->](0,0)--(2,2);\newframe
    \draw[->](0,0)--(3,3);\newframe
    \draw[->](0,0)--(4,4);
  \end{animateinline}
  \end{center}

\end{frame}
